function renderMainChart(data) { const ctx = document.getElementById('mainChart').getContext('2d'); if(charts.main) charts.main.destroy(); const mA = document.getElementById('metricA').value; const mB = document.getElementById('metricB').value; const sma = parseInt(document.getElementById('sma').value); const streams = data.filter(d => d["Minutes passées en stream"] > 0); const labels = streams.map(d => d._date.toLocaleDateString('fr-FR', {day:'2-digit', month:'short'})); const applySma = (key) => { const raw = streams.map(d => d[key]||0); if(sma <= 1) return raw; return raw.map((_, i) => { const slice = raw.slice(Math.max(0, i-sma+1), i+1); return slice.reduce((a,b)=>a+b,0) / slice.length; }); }; const datasets = [{ label: METRICS_DICT[mA].label, data: applySma(mA), borderColor: '#00e5ff', backgroundColor: 'rgba(0, 229, 255, 0.1)', fill: true, tension: 0.4, yAxisID: 'y' }]; if(mB !== 'none') datasets.push({ label: METRICS_DICT[mB].label, data: applySma(mB), borderColor: '#9146FF', borderDash: [5,5], tension: 0.4, yAxisID: 'y1' }); charts.main = new Chart(ctx, { type: 'line', data: { labels, datasets }, options: { onHover: (event, activeElements) => { if (activeElements && activeElements.length > 0) { const hoveredIndex = activeElements[0].index; if (typeof currentStreamList !== 'undefined' && currentStreamList && hoveredIndex >= 0 && hoveredIndex < currentStreamList.length) { if (currentStreamIndex !== hoveredIndex) { currentStreamIndex = hoveredIndex; updateStreamInspector(); } } } }, responsive: true, maintainAspectRatio: false, interaction: { mode: 'index', intersect: false }, plugins: { legend: { display: true, labels: { color: '#fff', font: { family: 'Orbitron', size: 10 } } } }, scales: { y: { grid: { color: 'rgba(255,255,255,0.05)' }, ticks: { color: '#00e5ff' } }, y1: { display: mB !== 'none', position: 'right', grid: { display: false }, ticks: { color: '#9146FF' } }, x: { ticks: { color: '#666' } } } } }); }
